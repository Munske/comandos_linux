echo "3-Remover usuário"
echo "4-Bloquear usuário"
echo "5-Desbloquear usuário"
read op
if [ $op -eq 1 ];then			-eq significa igual
	echo "Nome do novo usuário"
	read nome
	sudo adduser $nome
	sleep 5s
	clear
elif [ $op -eq 2 ];then
	sudo cat /etc/passwd
	sleep 5s
	clear
elif [ $op -eq 3 ];then
	echo "Nome do usuário que deve ser removido:"
	read nome
	sudo userdel -r $nome
	sleep 5s
	clear
elif [ $op -eq 4 ];then
	echo "Nome de usuário que deve ser bloqueado"
	read nome
	sudo passwd -l $nome
	sleep5
	clear
elif [ $op -eq 5 ];then
	echo "Nome de usuário que deseja desbloquear"
	read nome
	sudo passwd -u $nome
	sleep5
	clear
else
	echo "Operação invalida"
fi

(Sistema Operacional)



sudo ifconfig + placa de rede(ex:enp0s3) + down - desabilitar a placa de rede

sudo ifconfig + placa de rede(ex:enp0s3) + up   - ativar placa de rede

ifconfig lo  - mostra a interface virtual que faz comunicação entre o cliente e o server

sudo ifconfig + placa de rede(ex:enp0s3) + IP - definindo um Ip a placa de rede (fixando IP)

sudo ifconfig + placa de rede(ex:enp0s3) + IP netmask 255.255.255.0 - irá fixar o IP e a máscara de rede


ifconfig + placa de rede(ex:enp0s3) + netmask 255.255.255.0 - irá fixar a máscara da rede

netstat -a - irá exibir todas conexões rede

netstat -na - irá exibir todas conexões de rede com mais detalhes

netstat -lptu - exibe todas as portas que estão abertas

netstat -tulpn - exibe todas as portas que estão abertas com seus respectivos processos

netstat -sp -tcpp - exibe todas informações sobre todos os pacotes tpc

ifconfig + placa de rede + broadcast + o IP - irá definir o endereço de broadcast de uma interface

broadcast - É um método de transferência de mensagem para todos 
            os receptores simultaneamente

printenv - exibe as variáveis de ambiente

env - exibe as variáveis que foram exportadas

set - exibe os nomes das variáveis do shell

net share - exibe recursos compartilhados

start + sysvol - irá abrir a pasta solicitada
